    let currentLang = 'ko';

    // 번역 함수
    function translateMenu(lang) {
      // 섹션/카테고리명
      document.querySelectorAll('[data-key]').forEach(el => {
        const key = el.getAttribute('data-key');
        if (translations[key] && translations[key][lang]) {
          el.textContent = translations[key][lang];
        }
      });
    }

    // 언어 선택 이벤트
    document.querySelectorAll('.language-selector button').forEach(btn => {
      btn.onclick = function() {
        document.querySelectorAll('.language-selector button').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        currentLang = btn.getAttribute('data-lang');
        translateMenu(currentLang);
      }
    });

    // 메뉴명 최초 번역
    translateMenu(currentLang);

    // 팝업
    const overlay = document.querySelector('.overlay');
    const popup = document.querySelector('.popup');
    const popupDesc = popup.querySelector('.desc');
    const closeBtn = popup.querySelector('.close-btn');

    // 메뉴 클릭 이벤트
    document.querySelectorAll('.menu-item').forEach(item => {
      item.onclick = function() {
        const key = item.getAttribute('data-key');
        popupDesc.textContent = descriptions[key][currentLang];
        popup.classList.add('active');
        overlay.classList.add('active');
      }
    });
    // 팝업 닫기
    closeBtn.onclick = overlay.onclick = function() {
      popup.classList.remove('active');
      overlay.classList.remove('active');
    };

    // 언어 변경 시 팝업 내용도 갱신
    document.querySelectorAll('.language-selector button').forEach(btn => {
      btn.addEventListener('click', () => {
        if (popup.classList.contains('active')) {
          // 현재 열린 팝업의 key를 찾아서 설명 갱신
          const openedKey = Array.from(document.querySelectorAll('.menu-item')).find(item => {
            return descriptions[item.getAttribute('data-key')][currentLang] === popupDesc.textContent;
          })?.getAttribute('data-key');
          if (openedKey) popupDesc.textContent = descriptions[openedKey][currentLang];
        }
      });
    });
